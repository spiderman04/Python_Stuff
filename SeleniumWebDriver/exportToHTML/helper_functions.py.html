20   def setup_chrome():
21   
22       from selenium.webdriver.chrome.service import Service
23       # Override the Chrome Options
24       custom_options = webdriver.ChromeOptions()
25       custom_options.add_argument("--disable-notifications")
26       custom_options.add_argument("--start-fullscreen")
27       custom_options.add_argument("--disable-popup-blocking")
28       custom_options.add_argument("--disable-plugins")
29       custom_options.add_argument('--no-sandbox')
30       #custom_options.add_argument('--headless')
31       custom_options.add_argument('--disable-setuid-sandbox')
32       custom_options.add_argument('--disable-dev-shm-usage')
33       custom_options.add_argument('--disable-gpu')
34   
35       # Download the file to Project folder
36       print("Working Folder for Chrome: " + str(os.getcwd()))
37       custom_preferences = {"download.default_directory": os.getcwd()
38           , "plugins.always_open_pdf_externally": True}
39   
40       custom_options.add_experimental_option("prefs", custom_preferences)
41   
42       service_object = Service("C:\\WebDrivers\\chromedriver.exe")
43       driver = webdriver.Chrome(service=service_object, options=custom_options)
44       driver.implicitly_wait(20)
45       driver.maximize_window()
46   
47       return driver

75   def setup_firefox():
76   
77       from selenium.webdriver.firefox.service import Service
78       from selenium.webdriver import FirefoxProfile
79       from selenium.webdriver.common.desired_capabilities import DesiredCapabilities
80   
81       # Ensure we don't attempt to use the new geckodriver method (which
82       # isn't working for us. I _think_ selenium 2 defaults to old method,
83       # but just to make sure.
84       # capabilities = DesiredCapabilities.FIREFOX
85       # capabilities['marionette'] = False
86   
87       print("Working Folder for Firefox: " + str(os.getcwd()))
88       custom_options = webdriver.FirefoxOptions()
89       custom_options.set_preference("browser.helperApps.neverAsk.saveToDisk", "application/msword")
90       custom_options.set_preference("browser.helperApps.neverAsk.saveToDisk", "application/pdf")
91       custom_options.set_preference("browser.download.manager.showWhenStaring", False)
92       custom_options.set_preference("browser.download.folderList", 2)  # 0=Desktop, 1=Default location, 2=Custom
93       custom_options.set_preference("browser.download.dir", os.getcwd())
94       custom_options.set_preference("pdfjs.disabled", True)
95   
96       service_object = Service("C:\\WebDrivers\\geckodriver.exe")
97       driver = webdriver.Firefox(service=service_object, options=custom_options)
98   
99       driver.implicitly_wait(20)
100      driver.maximize_window()
101  
102      return driver
