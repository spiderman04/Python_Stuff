13   driver = webdriver.Chrome(executable_path="C:\\WebDrivers\\chromedriver.exe", service_args=None)
14   # driver.maximize_window()
15   
16   driver.get(constants.URL_MONEY_GAINERS)
17   
18   # XPATH AXES TYPES WE CAN USE
19   # self (current node), parent, child, ancestor, descendant, following, following-sibling, preceding,
20   # preceding-sibling, attribute, namespace, descendant-or-self, ancestor-or-self
21   
22   # self
23   self_node = driver.find_element(By.XPATH, "//a[contains(text(), 'India Tourism De')]/self::a").text
24   print("Self Node: " + self_node)
25   assert self_node == "India Tourism De"
26   
27   # parent
28   parent_node = driver.find_element(By.XPATH, "//a[contains(text(), 'India Tourism De')]/parent::td").text
29   print("Parent Node of Self: " + parent_node)
30   assert self_node == "India Tourism De"
31   
32   # child
33   child_nodes = driver.find_elements(By.XPATH, "//a[contains(text(), 'India Tourism De')]/ancestor::tr/child::td")
34   for child in child_nodes:
35       print("Child Node of Self: " + str(child.text))
36       #    print("Child Node of Self: " + driver.find_element(By.XPATH, child_node).text())
37   
38   # ancestor
39   ancestor_node = driver.find_element(By.XPATH, "//a[contains(text(), 'India Tourism De')]/ancestor::tr").text
40   print("Ancestor Node of Self: " + ancestor_node)
41   assert ancestor_node.__contains__("India Tourism De")
42   
43   # descendent
44   descendant_nodes = driver.find_elements(By.XPATH, "//a[contains(text(), 'India Tourism De')]/ancestor::tr/descendant::*")
45   print("Number of Descendant Nodes of Self: " + str(len(descendant_nodes)))
46   assert len(descendant_nodes) == 7, "Descendant nodes don't match 7"
47   
48   # following
49   following_nodes = driver.find_elements(By.XPATH, "//a[contains(text(), 'India Tourism De')]/ancestor::tr/following::*")
50   print("Number of Following Nodes of Self: " + str(len(following_nodes)))
51   assert len(following_nodes) > 0, "There are no following nodes of Self."
52   
53   # following-sibling
54   following_nodes = driver.find_elements(By.XPATH, "//a[contains(text(), 'India Tourism De')]/ancestor::tr/following-sibling::tr")
55   print("Number of Following-Sibling Nodes of Self: " + str(len(following_nodes)))
56   
57   # preceding
58   preceding_nodes = driver.find_elements(By.XPATH, "//a[contains(text(), 'India Tourism De')]/ancestor::tr/preceding::tr")
59   print("Number of Preceding Nodes of Self: " + str(len(following_nodes)))
60   
61   # preceding-sibling
62   preceding_nodes = driver.find_elements(By.XPATH, "//a[contains(text(), 'India Tourism De')]/ancestor::tr/preceding-sibling::tr")
63   print("Number of Preceding-Sibling Nodes of Self: " + str(len(following_nodes)))
64   
65   driver.close()
